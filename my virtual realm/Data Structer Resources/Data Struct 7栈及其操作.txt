###栈###
栈是一种只能在一端进行插入或删除的线性表，在主端进行的操作称为进栈（push）和出栈（pop）。我们使用数组来实现一个简单的栈。
#include <stdio.h>
#include <stdlib.h>

#define MAX_SIZE 100  // 定义栈的最大容量

typedef struct {
    int data[MAX_SIZE];  // 用于存储栈元素的数组
    int top;             // 栈顶指针
} Stack;

// 初始化栈
void initStack(Stack *s) {
    s->top = -1;  // 初始时栈为空
}

// 判断栈是否为空
int isEmpty(Stack *s) {
    return s->top == -1;
}

// 判断栈是否已满
int isFull(Stack *s) {
    return s->top == MAX_SIZE - 1;
}

// 入栈操作
void push(Stack *s, int value) {
    if (isFull(s)) {
        printf("Stack is full!\n");
        return;
    }
    s->data[++(s->top)] = value;  // 先移动指针再赋值
}

// 出栈操作
int pop(Stack *s) {
    if (isEmpty(s)) {
        printf("Stack is empty!\n");
        exit(1);  // 异常退出
    }
    return s->data[(s->top)--];  // 先取值再移动指针
}

// 获取栈顶元素
int peek(Stack *s) {
    if (isEmpty(s)) {
        printf("Stack is empty!\n");
        exit(1);
    }
    return s->data[s->top];
}
