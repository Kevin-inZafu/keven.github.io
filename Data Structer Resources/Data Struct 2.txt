1.编写函数，创建线性表(顺序表)
typedef struct
{
    int size, length;
    LELEMENT *element;
} ALIST;
#include <stdio.h>
#include <stdlib.h>
#include "AList.h"
int main()
{   ALIST a;
    AListCreate(&a);
    printf("%d %d %s\n", a.size, a.length, a.element ? "Valid" : "Error");
    free(a.element);
    return 0;}
请创建以void AListCreate(ALIST *list)为接口的函数用于创建顺序表
answer:
void AListCreate(ALIST *list)
{
list->element=malloc(AListInitSize*sizeof(LELEMENT));
    if(!list->element){
        exit(1);
    }
    list->length=0;
    list->size=AListInitSize;
}

4.本题要求实现一个函数，在顺序表的第i个位置插入一个新的数据元素e，插入成功后顺序表的长度加1，函数返回值为1；插入失败函数返回值为0；
函数的接口为int ListInsert(SqList &L,int i,ElemType e)
answer:
int ListInsert(SqList &L,int i,ElemType e)
{
    if((L.length==MAXSIZE)||(i>L.length+1)||(i<1))
        return 0;
     else
    {int real_i=i-1;
     int j=L.length;
     for(j;j>real_i;j--){
        L.elem[j]=L.elem[j-1];
     } L.elem[real_i]=e;
        L.length+=1;
        return 1;
    }
}

bool ListInsert(SqList *&L,int i,ElemType e);  //在顺序表第i个位置上插入一个结点，插入成功时，返回TRUE,否则返回FALSE.
answer：
bool ListInsert(SqList *&L,int i,ElemType e)
{
    if((L->length==MaxSize)||(i>L->length+1)||(i<1))
        return false;
     else
    {int real_i=i-1;
     int j=L->length;
     for(j;j>real_i;j--){
        L->data[j]=L->data[j-1];
     } L->data[real_i]=e;
        L->length+=1;
        return true;
    }
}

answer:
int GetMax(SqList L)
{
    if(L.length==0)
        return 0;
    else
    {int max=L.elem[0];
        for(int j=1;j<L.length;j++)
            if(L.elem[j]>max)
                max=L.elem[j];

    }
